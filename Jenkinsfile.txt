pipeline {
    agent any

    environment {
        SONARQUBE_URL = 'http://your.sonarqube.server'
        SONARQUBE_TOKEN = 'squ_912f8bb937d3622d051f807758ae7fc0d78befdb'
    }

    stages {
        stage('GIT') {
            steps {
                git branch: 'main', url: 'https://github.com/oussamakaddech/atelierdevops.git'
            }
        }

        stage('Compile') {
            steps {
                sh 'mvn clean compile'
            }
        }

        stage('SonarQube Analysis (via Docker)') {
            steps {
                script {
                    // Run SonarQube analysis using Docker container
                    sh """
                    docker run --rm \
                    -v \$(pwd):/usr/src \
                    -v /root/.m2:/root/.m2 \
                    sonarsource/sonar-scanner-cli \
                    -Dsonar.projectKey=your_project_key \
                    -Dsonar.sources=src \
                    -Dsonar.host.url=${SONARQUBE_URL} \
                    -Dsonar.login=${SONARQUBE_TOKEN}
                    """
                }
            }
        }

        stage('Deploy') {
            steps {
                sh 'mvn clean deploy -Dmaven.test.skip=true'
            }
        }
    }

    post {
        always {
            echo 'Pipeline finished.'
        }
        success {
            echo 'Pipeline succeeded.'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
