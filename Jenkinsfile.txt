pipeline {
    agent any

    tools {
        maven 'M2_HOME'
    }

    environment {
        SONARQUBE_TOKEN = 'squ_912f8bb937d3622d051f807758ae7fc0d78befdb' // Remplacez par votre token SonarQube ici
    }

    stages {
        stage('GIT') {
            steps {
                git branch: 'main', url: 'https://github.com/oussamakaddech/atelierdevops.git'
            }
        }

        stage('Stage Compile') {
            steps {
                sh 'mvn clean compile'
            }
        }

        stage('SonarQube Analysis') {
            steps {
                sh "mvn sonar:sonar -Dsonar.token=${SONARQUBE_TOKEN} -Dmaven.test.skip=true"
            }
        }

        stage('Deploy to Nexus') {
            steps {
                script {
                    sh 'mvn clean deploy -Dmaven.test.skip=true'
                }
            }
        }
    }

    post {
        always {
            echo 'Pipeline finished.'
        }
        success {
            echo 'Pipeline succeeded.'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
